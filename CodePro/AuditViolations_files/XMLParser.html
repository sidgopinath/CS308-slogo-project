76    /**

83    public static void clean(Node node) {
84       NodeList childNodes = node.getChildNodes();
85       for (int n = childNodes.getLength() - 1; n >= 0; n--) {
86          Node child = childNodes.item(n);
87          short nodeType = child.getNodeType();
88          if (nodeType == Node.ELEMENT_NODE) {
89             clean(child);
90          } else if (nodeType == Node.TEXT_NODE) {
91             String trimmedNodeVal = child.getNodeValue().trim();
92             if (trimmedNodeVal.length() == 0) {
93                node.removeChild(child);
94             } else {
95                child.setNodeValue(trimmedNodeVal);
96             }
97          } else if (nodeType == Node.COMMENT_NODE) {
98             node.removeChild(child);
99          }
100       }
